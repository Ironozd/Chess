26 template<length_t D, typename T, qualifier Q, typename I>
27 GLM_FUNC_QUALIFIER mat<D, D, T, Q> computeCovarianceMatrix(I const& b, I const& e)
28 {
29 glm::mat<D, D, T, Q> m(0);
30
31 size_t cnt = 0;
32 for(I i = b; i != e; i++)
33 {
34 vec<D, T, Q> const& v = *i;
35 for(length_t x = 0; x < D; ++x)
36 for(length_t y = 0; y < D; ++y)
37 m[x][y] += static_cast<T>(v[x] * v[y]);
38 cnt++;
39 }
40 if(cnt > 0)
41 m /= static_cast<T>(cnt);
42
43 return m;
44 }

77 template<typename T>
78 GLM_FUNC_QUALIFIER static T pythag(T const& a, T const& b) {
79 static const T epsilon = static_cast<T>(0.0000001);
80 T absa = glm::abs(a);
81 T absb = glm::abs(b);
82 if(absa > absb) {
83 absb /= absa;
84 absb *= absb;
85 return absa * glm::sqrt(static_cast<T>(1) + absb);
86 }
87 if(glm::equal<T>(absb, 0, epsilon)) return static_cast<T>(0);
88 absa /= absb;
89 absa *= absa;
90 return absb * glm::sqrt(static_cast<T>(1) + absa);
91 }

95 template<length_t D, typename T, qualifier Q>
96 GLM_FUNC_QUALIFIER unsigned int findEigenvaluesSymReal
97 (
98 mat<D, D, T, Q> const& covarMat,
99 vec<D, T, Q>& outEigenvalues,
100 mat<D, D, T, Q>& outEigenvectors
101 )
102 {
103 using _internal_::transferSign;
104 using _internal_::pythag;
105
106 T a[D * D]; // matrix -- input and workspace for algorithm (will be changed inplace)
107 T d[D]; // diagonal elements
108 T e[D]; // off-diagonal elements
109
110 for(length_t r = 0; r < D; r++)
111 for(length_t c = 0; c < D; c++)
112 a[(r) * D + (c)] = covarMat[c][r];
113
114 // 1. Householder reduction.
115 length_t l, k, j, i;
116 T scale, hh, h, g, f;
117 static const T epsilon = static_cast<T>(0.0000001);
118
119 for(i = D; i >= 2; i--)
120 {
121 l = i - 1;
122 h = scale = 0;
123 if(l > 1)
124 {
125 for(k = 1; k <= l; k++)
126 {
127 scale += glm::abs(a[(i - 1) * D + (k - 1)]);
128 }
129 if(glm::equal<T>(scale, 0, epsilon))
130 {
131 e[i - 1] = a[(i - 1) * D + (l - 1)];
132 }
133 else
134 {
135 for(k = 1; k <= l; k++)
136 {
137 a[(i - 1) * D + (k - 1)] /= scale;
138 h += a[(i - 1) * D + (k - 1)] * a[(i - 1) * D + (k - 1)];
139 }
140 f = a[(i - 1) * D + (l - 1)];
141 g = ((f >= 0) ? -glm::sqrt(h) : glm::sqrt(h));
142 e[i - 1] = scale * g;
143 h -= f * g;
144 a[(i - 1) * D + (l - 1)] = f - g;
145 f = 0;
146 for(j = 1; j <= l; j++)
147 {
148 a[(j - 1) * D + (i - 1)] = a[(i - 1) * D + (j - 1)] / h;
149 g = 0;
150 for(k = 1; k <= j; k++)
151 {
152 g += a[(j - 1) * D + (k - 1)] * a[(i - 1) * D + (k - 1)];
153 }
154 for(k = j + 1; k <= l; k++)
155 {
156 g += a[(k - 1) * D + (j - 1)] * a[(i - 1) * D + (k - 1)];
157 }
158 e[j - 1] = g / h;
159 f += e[j - 1] * a[(i - 1) * D + (j - 1)];
160 }
161 hh = f / (h + h);
162 for(j = 1; j <= l; j++)
163 {
164 f = a[(i - 1) * D + (j - 1)];
165 e[j - 1] = g = e[j - 1] - hh * f;
166 for(k = 1; k <= j; k++)
167 {
168 a[(j - 1) * D + (k - 1)] -= (f * e[k - 1] + g * a[(i - 1) * D + (k - 1)]);
169 }
170 }
171 }
172 }
173 else
174 {
175 e[i - 1] = a[(i - 1) * D + (l - 1)];
176 }
177 d[i - 1] = h;
178 }
179 d[0] = 0;
180 e[0] = 0;
181 for(i = 1; i <= D; i++)
182 {
183 l = i - 1;
184 if(!glm::equal<T>(d[i - 1], 0, epsilon))
185 {
186 for(j = 1; j <= l; j++)
187 {
188 g = 0;
189 for(k = 1; k <= l; k++)
190 {
191 g += a[(i - 1) * D + (k - 1)] * a[(k - 1) * D + (j - 1)];
192 }
193 for(k = 1; k <= l; k++)
194 {
195 a[(k - 1) * D + (j - 1)] -= g * a[(k - 1) * D + (i - 1)];
196 }
197 }
198 }
199 d[i - 1] = a[(i - 1) * D + (i - 1)];
200 a[(i - 1) * D + (i - 1)] = 1;
201 for(j = 1; j <= l; j++)
202 {
203 a[(j - 1) * D + (i - 1)] = a[(i - 1) * D + (j - 1)] = 0;
204 }
205 }
206
207 // 2. Calculation of eigenvalues and eigenvectors (QL algorithm)
208 length_t m, iter;
209 T s, r, p, dd, c, b;
210 const length_t MAX_ITER = 30;
211
212 for(i = 2; i <= D; i++)
213 {
214 e[i - 2] = e[i - 1];
215 }
216 e[D - 1] = 0;
217
218 for(l = 1; l <= D; l++)
219 {
220 iter = 0;
221 do
222 {
223 for(m = l; m <= D - 1; m++)
224 {
225 dd = glm::abs(d[m - 1]) + glm::abs(d[m - 1 + 1]);
226 if(glm::equal<T>(glm::abs(e[m - 1]) + dd, dd, epsilon))
227 break;
228 }
229 if(m != l)
230 {
231 if(iter++ == MAX_ITER)
232 {
233 return 0; // Too many iterations in FindEigenvalues
234 }
235 g = (d[l - 1 + 1] - d[l - 1]) / (2 * e[l - 1]);
236 r = pythag<T>(g, 1);
237 g = d[m - 1] - d[l - 1] + e[l - 1] / (g + transferSign(r, g));
238 s = c = 1;
239 p = 0;
240 for(i = m - 1; i >= l; i--)
241 {
242 f = s * e[i - 1];
243 b = c * e[i - 1];
244 e[i - 1 + 1] = r = pythag(f, g);
245 if(glm::equal<T>(r, 0, epsilon))
246 {
247 d[i - 1 + 1] -= p;
248 e[m - 1] = 0;
249 break;
250 }
251 s = f / r;
252 c = g / r;
253 g = d[i - 1 + 1] - p;
254 r = (d[i - 1] - g) * s + 2 * c * b;
255 d[i - 1 + 1] = g + (p = s * r);
256 g = c * r - b;
257 for(k = 1; k <= D; k++)
258 {
259 f = a[(k - 1) * D + (i - 1 + 1)];
260 a[(k - 1) * D + (i - 1 + 1)] = s * a[(k - 1) * D + (i - 1)] + c * f;
261 a[(k - 1) * D + (i - 1)] = c * a[(k - 1) * D + (i - 1)] - s * f;
262 }
263 }
264 if(glm::equal<T>(r, 0, epsilon) && (i >= l))
265 continue;
266 d[l - 1] -= p;
267 e[l - 1] = g;
268 e[m - 1] = 0;
269 }
270 } while(m != l);
271 }
272
273 // 3. output
274 for(i = 0; i < D; i++)
275 outEigenvalues[i] = d[i];
276 for(i = 0; i < D; i++)
277 for(j = 0; j < D; j++)
278 outEigenvectors[i][j] = a[(j) * D + (i)];
279
280 return D;
281 }
